\begin{eqnarray*}
\frac{n!}{r!(n-r)!}&=&\frac{(n-1)!}{r!(n-1-r)!}+\frac{(n-1)!}{(r-1)!((n-1)-(r-1))!}\\
&=&\frac{(n-1)!}{r!(n-1-r)!}+\frac{(n-1)!}{(r-1)!(n-r)!}\\
&=&...\\
&=&\frac{(n-r+r)(n-1)!}{r(r-1)!(n-r)(n-r-1)!}\\
&=&\frac{(n)(n-1)!}{r(r-1)!(n-r)(n-r-1)!}\\
&=&\frac{n!}{r!(n-r)!}\\
\end{eqnarray*}
